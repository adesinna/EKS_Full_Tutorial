2. Install Nginx on the Jenkins Instance

sudo apt update
sudo apt upgrade

sudo apt install nginx

systemctl status nginx


sudo vim /etc/nginx/sites-available/ci.shananadevops.xyz   # point the record to the instance


# paste the code below

upstream jenkins{
    server 127.0.0.1:8080;
}

server{
    listen      80;
    server_name ci.shananadevops.xyz;

    access_log  /var/log/nginx/jenkins.access.log;
    error_log   /var/log/nginx/jenkins.error.log;

    proxy_buffers 16 64k;
    proxy_buffer_size 128k;

    location / {
        proxy_pass  http://jenkins;
        proxy_next_upstream error timeout invalid_header http_500 http_502 http_503 http_504;
        proxy_redirect off;

        proxy_set_header    Host            $host;
        proxy_set_header    X-Real-IP       $remote_addr;
        proxy_set_header    X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header    X-Forwarded-Proto https;
    }

}


# create a link

sudo ln -s /etc/nginx/sites-available/ci.shananadevops.xyz /etc/nginx/sites-enabled/


# test and verify

sudo nginx -t

# restart your nginx

sudo systemctl restart nginx


# note that port 80 and 443 should be opened, close 8080

# cert bot should be for install for tls

sudo apt install certbot python3-certbot-nginx -y


# get certificate

sudo  certbot --nginx

sudo certbot renew --dry-run # to renew the certificate

sudo nginx -t

sudo systemctl restart nginx
